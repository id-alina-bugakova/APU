------------------------------------- STATES
0     STATE_OFF,                      // ВСУ выключена
1     STATE_IDLE,                     // Ожидание команды запуска
2     STATE_TEST,                     // Автоматическая диагностика системы
3     STATE_START,                    // Ручное включение ВСУ: электростартер, зажигание, открытие 
                                      //     клапанов, включение агрегатов
4     STATE_AUTO_START,               // Автоматическое включение в воздухе (при отказе бортовой 
                                      //     сети)
5     STATE_EMERGENCY_START,          // Аварийный запуск в обход ВСЕХ мер безопасности (пожар, 
                                      //     большая высота); ограниченные обороты + запрет отбора 
                                      //     воздуха
6     STATE_IDLE_RUN,                 // Работа на малых оборотах (без нагрузки)
7     STATE_BLEED,                    // Работа под нагрузкой (СКВ)
8     STATE_GEN,                      // Работа под нагрузкой (генератор)
9     STATE_LOAD,                     // Работа под нагрузкой (СКВ + генератор)
10    STATE_MPU_START,                // Работа на максимальных оборотах (для запуска МСУ)
11    STATE_IDLE_RUN_LIMITED,         // Аварийная работа на малых оборотах
12    STATE_GEN_LIMITED,              // Аварийная работа на генератор ВСУ
13    STATE_MPU_START_LIMITED,        // Аварийная работа на запуск МСУ
14    STATE_COOLDOWN,                 // Охлаждение: снижение оборотов в течение 3 минут без нагрузки
15    STATE_NORMAL_SHUTDOWN,          // Отключение при оборотах N < 5%
16    STATE_APU_FIRE,                 // Аварийный режим - пожар ВСУ: перекрытие клапанов, выключение 
                                      //     и запрет на повторное включение
17    STATE_EGT_OVER_LIMIT,           // Аварийный режим - перегрев турбины: остановка, отсечка 
                                      //     топлива
18    STATE_OVSPEED,                  // Аварийный режим - разнос ротора: остановка, отсечка топлива
19    STATE_FLAME_OUT,                // Аварийный режим - потеря пламени: попытка повторного 
                                      //     зажигания до 2 раз
20    STATE_RELIGHT,                  // Аварийный режим - попытка повторного зажигания
21    STATE_EMERGENCY_SHUTDOWN        // Аварийное отключение: быстрая остановка ВСУ


------------------------------------- EVENTS
0     EVENT_POWER_ON,                 // Подача питания ВСУ
1     EVENT_START_TEST,               // Начало диагностики
2     EVENT_TEST_COMPLETED,           // Диагностика завершена
3     EVENT_START_CMD,                // Получена команда ручного запуска
4     EVENT_AUTO_START_CMD,           // Получена команда автоматического запуска
5     EVENT_EMERGENCY_START_CMD,      // Получена команда аварийного запуска
6     EVENT_START_SUCCESS,            // Успешный запуск
7     EVENT_START_FAILURE,            // Неудачный запуск
8     EVENT_START_ABORTED,            // Запуск прерван
9     EVENT_BLEED_ON_CMD,             // Получена команда включения отбора воздуха в СКВ
10    EVENT_GEN_ON_CMD,               // Получена команда включения генератора
11    EVENT_MPU_ON_CMD,               // Получена команда запуска МСУ
12    EVENT_BLEED_OFF_CMD,            // Получена команда выключения отбора воздуха в СКВ
13    EVENT_GEN_OFF_CMD,              // Получена команда выключения генератора
14    EVENT_MPU_OFF_CMD,              // Получена команда завершения запуска МСУ
15    EVENT_NORMAL_SHUTDOWN_CMD,      // Получена команда выключения
16    EVENT_COOLDOWN_TIMEOUT,         // Таймер охлаждения истек
17    EVENT_APU_STOPPED,              // ВСУ остановлена (менее 5% оборотов)
18    EVENT_POWER_OFF,                // Отключено питание
19    EVENT_FIRE_DETECTED,            // Обнаружен пожар в отсеке ВСУ
20    EVENT_OVHEAT_DETECTED,          // Произошел перегрев
21    EVENT_OVSPEED_DETECTED,         // Произошло превышение скорости вращения двигателя
22    EVENT_FLAME_WENT_OUT,           // Пропал факел в камере сгорания
23    EVENT_RELIGHT_ATTEMPT_CMD,      // Получена команда произвести попытку повторного зажигания
24    EVENT_FLAME_RESTORED,           // После повторного зажигания пламя появилось
25    EVENT_EMERGENCY_SHUTDOWN_CMD    // Получена команда аварийного выключения
